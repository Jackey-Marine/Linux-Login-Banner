function printBar() {
	input=$1
	roundedInput=$(awk -v x=$input 'BEGIN{printf "%3.0f\n", x/2}')
	output=""
	for ((i=1;i<=$roundedInput;i++))
	do
		output=$output#
	done
	for ((i=1;i<=$((50 - $roundedInput));i++))
	do
		output+="-"
	done
	echo $output
}
